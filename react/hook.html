ref 的使用：

无论子组件是函数组件还是 class 组件，父组件（函数或class）的写法不变

  1. 子组件

    • hook

      import React, { forwardRef, useImperativeHandle } from "react";

const Child = forwardRef((props, ref) => {
	useImperativeHandle(ref, () => ({ sayHi }));
  
	const sayHi = () => {
		console.log("我是子组件方法");
	};
	return <div>我是子组件</div>;
});

export default Child;

    • class

      import React, { Component } from "react";

class Child extends Component {
	constructor(props) {
		super(props);
	}
	sayHi = () => {
		console.log("我是子组件方法");
	};
	render() {
		return <div>我是子组件</div>;
	}
}

export default Child;

  2. 父组件

    • hook

      import React, { useRef } from "react";
import Child from "./Child";

const Parent = () => {
	const childRef = useRef(null);

	const handleClick = () => {
    childRef.current.sayHi() // 我是子组件方法
	};
	return (
		<>
			<div>我是父组件</div>
			<button onClick={() => handleClick()}>点击</button>
			<Child ref={childRef} />
		</>
	);
};

export default Parent;

    • class

      import React, { Component, createRef } from "react";
import Child from "./Child";

class Parent extends Component {
	constructor(props) {
		super(props);
		// this.childRef = createRef(); 第一种方式
    	this.childRef = null
	}
	handleClick = () => {
        // this.childRef.current.sayHi() 第一种方式
        this.childRef.sayHi()
	}
	render() {
		return (
			<>
				<div>我是父组件</div>
				<button onClick={() => this.handleClick()}>点击</button>
            	// <Child ref={this.childRef/> 第一种方式
                <Child ref={(ref) => this.childRef = ref} />
			</>
		);
	}
}

export default Parent;
