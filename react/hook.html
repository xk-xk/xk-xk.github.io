# useState

useState 用来管理组件自身的状态

import React, { useState } from 'react'

const App = () => {
    const [count, setCount] = useState(0)
    
    return (
    	<div>
        	{count}
            <button onClick={() => setCount(count+1)}>+1</button>
        </div>
    )
}

export default App

# useRef

ref 的使用：

无论子组件是函数组件还是 class 组件，父组件（函数或class）的写法不变

  1. 子组件

    • hook

      import React, { forwardRef, useImperativeHandle } from "react";

const Child = forwardRef((props, ref) => {
	useImperativeHandle(ref, () => ({ sayHi }));
  
	const sayHi = () => {
		console.log("我是子组件方法");
	};
	return <div>我是子组件</div>;
});

export default Child;

    • class

      import React, { Component } from "react";

class Child extends Component {
	constructor(props) {
		super(props);
	}
	sayHi = () => {
		console.log("我是子组件方法");
	};
	render() {
		return <div>我是子组件</div>;
	}
}

export default Child;

  2. 父组件

    • hook

      import React, { useRef } from "react";
import Child from "./Child";

const Parent = () => {
	const childRef = useRef(null);

	const handleClick = () => {
    childRef.current.sayHi() // 我是子组件方法
	};
	return (
		<>
			<div>我是父组件</div>
			<button onClick={() => handleClick()}>点击</button>
			<Child ref={childRef} />
		</>
	);
};

export default Parent;

    • class

      import React, { Component, createRef } from "react";
import Child from "./Child";

class Parent extends Component {
	constructor(props) {
		super(props);
		// this.childRef = createRef(); 第一种方式
    	this.childRef = null
	}
	handleClick = () => {
        // this.childRef.current.sayHi() 第一种方式
        this.childRef.sayHi()
	}
	render() {
		return (
			<>
				<div>我是父组件</div>
				<button onClick={() => this.handleClick()}>点击</button>
            	// <Child ref={this.childRef/> 第一种方式
                <Child ref={(ref) => this.childRef = ref} />
			</>
		);
	}
}

export default Parent;

# useEffect

useEffect 用来实现组件的生命周期（函数式组件不继承 React.Component，因此不存在生命周期的方法）

接收两个参数，第一个参数是函数，用来执行具体的操作，第二个为数组，表明哪些数据变化会引起函数的执行，

传空 []则代表只执行一次，相当于 componentDidMount

import React, { useEffect } from 'react'

const App = () => {
    useEffect(() => {
        // componentDidMount
    }, [])
    
    useEffect(() => {
        return () => {
            // componentWillUnmount
            timer && clearInterval(timer)
        }
    }, [])
    
    useEffect(() => {
        // componentDidUpdate
    }, [a, b])
}

export default App

# useReducer

useReducer是useState的替代方案，同样是用来处理状态数据，更适合用来处理复杂状态逻辑

参考官网示例：

function init(initialCount) {
  return {count: initialCount};
}

function reducer(state, action) {
  switch (action.type) {
    case 'increment':
      return {count: state.count + 1};
    case 'decrement':
      return {count: state.count - 1};
    case 'reset':
      return init(action.payload);
    default:
      throw new Error();
  }
}

function Counter({initialCount}) {
  const [state, dispatch] = useReducer(reducer, initialCount, init);
  return (
    <>
      Count: {state.count}
      <button
        onClick={() => dispatch({type: 'reset', payload: initialCount})}>
        Reset
      </button>
      <button onClick={() => dispatch({type: 'decrement'})}>-</button>
      <button onClick={() => dispatch({type: 'increment'})}>+</button>
    </>
  );
}

# useContext

useContext 用来解决从父组件向子组件层层 props 传递数据的问题

父组件创建 context：

import React from 'react';
import { Itodo } from './typings';

interface ITodoContext {
    removeTodo: (todo: Itodo) => void;
}
const defaultContext = {
    removeTodo: () => {},
};

export const TodoContext = React.createContext<ITodoContext>(defaultContext)

const TodoList: React.FC = () => {
    
    const removeTodo = (id: number) => {
        // do something
    }
    const contextValue = {
        removeTodo,
    }
    return (
    	<div>
            <TodoContext.Provider value={ContextValue}>
            	...
            </TodoContext.Provider>
        </div>
    )
}

export default TodoList

子组件引用 context ：

import React, { useContext } from 'react';
import { TodoContext } from '../index';
interface ITodoInfo {
    removeTodo: (id: number) => void;
}

const TdItem = ({ todo }) => {
    const todoInfo: ITodoInfo = useContext(TodoContext)
    
    return (
    	<div>
        	<button onClick={() => todoInfo.removeTodo(todo.id)}>删除</button>
        </div>
    )
}

export default TdItem;

# useDebounce

使用 useEffect 和自定义hook实现一个函数防抖

import { useEffect, useState } from 'react'

function useDebounce(value: any, delay = 300) {
    const [debouncedValue, setDebouncedValue] = useState(value)
    
    useEffect(() => {
        const handler = window.setTimeout(() => {
            setDebouncedValue(value)
        }, delay)
        
        return () => {
            clearTimeout(handler)
        }
    }, [value, delay])
    
    return debouncedValue
}

# useClickOutside

点击元素之外的区域触发

import { RefObject, useEffect } from "react";

/**
 * 点击 dom 外的区域
 * @param refs ref数组
 * @param callback 回调函数
 */
export const useClickOutside = (refs: RefObject<HTMLElement>[], callback: Function) => {
    useEffect(() => {
        const listener = (event: MouseEvent) => {
            try {
                refs.forEach((ref) => {
                    if (!ref.current || ref.current?.contains(event.target as HTMLElement)) {
                        throw new Error();
                    }
                });
                callback(event);
            } catch (error) {}
        };

        document.addEventListener("click", listener);
        return () => {
            document.removeEventListener("click", listener);
        };
    }, [refs, callback]);
};

使用

useClickOutside([ref1, ref2], () => dosomething())
