Get.jsx

class Get extends React.Component {
    state = {
        data: null,
        component: null
    }
    async componentDidMount() {
        const { url, delay, children, loading, error } = this.props
        console.log(children)
        try {
            const result = await axios(url)
            console.log(result);
            this.setState({
                component: loading
            })
            this.setState({
                data: result.data
            }, () => {
                setTimeout(() => {
                    this.setState({
                        component: children(this.state.data)
                    })
                }, delay || 0)
            })
        } catch (e) {
            this.setState({
                component: error
            })
        }
    }
    render() {
        return this.state.component
    }
}

export default Get

Post.jsx

import qs from 'qs'
class Post extends React.Component {
    state = {
        component: null,
        data: null
    }
    async componentDidMount() {
        const { url, data, delay, loading, error, children, dataOperate } = this.props
        try {
            this.setState({
                component: loading
            })
            let result = await axios.post(url, qs.stringify(data))
            this.setState({
                data: dataOperate(result.data)
            }, () => {
                setTimeout(() => {
                    this.setState({
                        component: children(result.data)
                    })
                }, delay || 0)
            })
        } catch (e) {
            this.setState({
                component: error
            })
        }

    }
    render() {
        return this.state.component
    }
}

export default Post

import HTTP from '../utils/HTTP'
class Students extends React.Component {
    state = {
        postParams: {
            grade: 2
        }
    }
    dataOperate(data) {
        return data.map(item => {
            item.key = new Date().getTime()
            return item
        })
    }
    render() {
        return (
            <div>
                <table border='1'>
                    <thead>
                        <tr>
                            <th>Student Id</th>
                            <th>Student name</th>
                            <th>Grage No.</th>
                        </tr>
                    </thead>
                    <tbody>
                        <HTTP.Get
                            delay={1000}
                            url="http://localhost:8080/getStudents"
                            loading={
                                <tr align='center'>
                                    <td colSpan="3">Data is loading...</td>
                                </tr>
                            }
                            error={
                                <tr align='center'>
                                    <td colSpan="3">Some error occured!</td>
                                </tr>
                            }
                         >
                            {
                                (data) => {
                                    return data.map(item => (
                                        <tr key={item.id}>
                                            <td>{item.id}</td>
                                            <td>{item.name}</td>
                                            <td>{item.grade}</td>
                                        </tr>
                                    ))
                                }
                            }
                        </HTTP.Get>
                    </tbody>
                </table>
                <table border='1'>
                    <thead>
                        <tr>
                            <th>Student Id</th>
                            <th>Student name</th>
                            <th>Grage No.</th>
                            <th>Grage Key.</th>
                        </tr>
                    </thead>
                    <tbody>
                        <HTTP.Post
                            data={this.state.postParams}
                            delay={1000}
                            url="http://localhost:8080/getGradeStudents"
                            loading={
                                <tr align='center'>
                                    <td colSpan="4">Data is loading...</td>
                                </tr>
                            }
                            error={
                                <tr align='center'>
                                    <td colSpan="4">Some error occured!</td>
                                </tr>
                            }
                            dataOperate={this.dataOperate}
                        >
                            {
                                (data) => {
                                    return data.map(item => (
                                        <tr key={item.id}>
                                            <td>{item.id}</td>
                                            <td>{item.name}</td>
                                            <td>{item.grade}</td>
                                            <td>{item.key}</td>
                                        </tr>
                                    ))
                                }
                            }
                        </HTTP.Post>
                    </tbody>
                </table>
            </div>
        )
    }
}

export default Students
