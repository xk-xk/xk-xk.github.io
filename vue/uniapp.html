# 1、保存图片到手机相册

	/**
	 * 保存图片到手机相册
	 * @param {string} imgUrl 图片链接
	 * @param {function} successCallback 成功回调函数
	 * @param {function} errorCallback 失败回调函数
	 */
	saveImageToPhotosAlbum(imgUrl, successCallback, errorCallback) {
	  uni.showLoading({
	    title: '保存中...'
	  });
	  const url = wx.env.USER_DATA_PATH + "/" + new Date().getTime().toString() + imgUrl.slice(imgUrl.lastIndexOf('.'))
	  console.log('filePath', url)
	  uni.downloadFile({
	    url: imgUrl,
		filePath: url,
	    success: function(res) {
		console.log('下载成功', res)
	      uni.saveImageToPhotosAlbum({
	        filePath: url,
	        success: function(result) {
			console.log('保存成功', result)
	          uni.hideLoading();
	          uni.showToast({
	            title: '保存成功',
	            icon: 'success',
	            duration: 2000
	          });
	          if (typeof successCallback === 'function') {
	            successCallback();
	          }
	        },
	        fail: function(error) {
			  console.log('保存失败', error)
	          uni.hideLoading();
	          uni.showToast({
	            title: '保存失败',
	            icon: 'none',
	            duration: 2000
	          });
	          if (typeof errorCallback === 'function') {
	            errorCallback(res);
	          }
	        }
	      });
	    },
	    fail: function(res) {
		  console.log('下载图片失败', res)
	      uni.hideLoading();
	      uni.showToast({
	        title: '下载图片失败',
	        icon: 'none',
	        duration: 2000
	      });
	      if (typeof errorCallback === 'function') {
	        errorCallback(res);
	      }
	    }
	  });
	},

# 2、判断当前环境是否为真机

	isRealPhone() {
		let systemInfo = uni.getSystemInfoSync()
		let platform = systemInfo.platform
		if (platform === 'ios' || platform === 'android') {
			return true
		} else {
			return false
		}
	}
